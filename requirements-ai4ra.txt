# Optional extras for the AI4RA-aligned tools.
#
# Cambium stays stdlib-first. Every tool below degrades gracefully without these:
# it still runs and writes a useful result, and tells you which path it used.
# Install these only if you want the stronger, standards-backed behavior.
#
#   pip install -r requirements-ai4ra.txt
#
# jsonschema   -- full JSON Schema validation for tools/rules_handoff.py
#                 (Vandalizer -> Cambium handoff). Without it, a minimal stdlib
#                 required-keys check runs instead.
# frictionless -- validate the FAIR Data Package descriptor from
#                 tools/fair_descriptor.py against the open data-package profile.
#                 Without it, the descriptor is still written, just not externally validated.
# prov         -- emit a W3C PROV document from tools/tamper_record.py --format prov.
#                 Without it, --format prov falls back to json and says so.
# pip-tools    -- compile pinned, hash-locked requirements for reproducible runs
#                 (AI4RA Pillar: Reproducibility). A workflow aid, not imported by any tool.

frictionless==5.18.0
